**修改标签的属性值**

通过obj.setAttribute('属性','属性值')，修改标签的属性值

通过obj.getAttribute('属性')，来输出标签的属性值

通过this.属性 = 'xxx'，输出、修改标签的属性值

在JS中如果想修改标签的class属性，格式为obj.className = 'xxx'

**通过定义函数的形式，来获取标签对象**

```javascript
function $(id){
                var obj = document.getElementById(id);
                return obj;
            }
```

鼠标悬浮方法:onmouseover

鼠标移开方法:onmouseout

```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <meta http-equiv="X-UA-Compatible" content="ie=edge">
    <title>Document</title>
</head>
<body>
    <div class="box">
        <img id='service' src="./images/service.png" alt="">
    </div>
    <script type="text/javascript">
        window.onload = function(){
            var imgobj = document.getElementById('service');
            console.log(imgobj);
            
            // onmouseover表示鼠标浮动到对象时执行的方法
            imgobj.onmouseover = function(){
                // this表示当前对象imgobj
                console.log(this);

                // 通过对象.setAttribute('src','./images/service_hover.png');设置属性
                imgobj.setAttribute('src','./images/service_hover.png');
                
                // 或是通过imgobj.getAttribute('src');获取对象的属性值
                console.log(imgobj.getAttribute('src'));

                // file:///E:/workspace/wxPythonCode/m5/JS/images/service_hover.png打印的是绝对路径
                console.log(this.src);

                // 也可以直接使用this.属性名 = 'xxx'的方式去设置属性值
                this.src = './images/service_hover.png';
            }
            
            // onmouseout表示鼠标移开时执行的方法
            imgobj.onmouseout = function(){
                // imgobj.setAttribute('src','./images/service.png');
                this.src = './images/service.png';
            }
        }
    </script>
</body>
</html>
```



修改标签的属性值练习

隐藏或显示盒子

**方法一：修改标签的CSS属性，将其设置display设置为none**

```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <meta http-equiv="X-UA-Compatible" content="ie=edge">
    <title>Document</title>
    <style type="text/css">
        #box{
            width: 100px;
            height: 100px;
            background-color: red;
        }
    </style>
</head>
<body>
    <button id="btn">隐藏</button>
    <div id="box">

    </div>
    <script type="text/javascript">
        window.onload = function(){
            function $(id){
                obj = document.getElementById(id);
                return obj;
            };

            var isShow = true;
            $('btn').onclick = function(){
                if(isShow){
                    $('box').style.display = 'none';
                    $('btn').innerText = '显示';
                    isShow = false;
                }else{
                    $('box').style.display = 'block';
                    $('btn').innerText = '隐藏';
                    isShow = true;
                }
            }
        }        
    </script>
</body>
</html>
```



**方法二：修改标签的class，将其继承于其他类**

```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <meta http-equiv="X-UA-Compatible" content="ie=edge">
    <title>Document</title>
    <style type="text/css">
        .box{
            width: 100px;
            height: 100px;
            background-color: red;
        }
        .none{
            display: none;
        }
    </style>
</head>
<body>
    <button id="btn">隐藏</button>
    <div id='box' class="box"></div>

    <script type="text/javascript">
        window.onload = function(){
            function $(id){
                var obj = document.getElementById(id);
                return obj;
            }
            
            // 第二种方式是修改标签的class属性，在js中修改class属性格式为className驼峰式
            var isShow = true;
            $('btn').onclick = function(){
                console.log($('box'));
                if(isShow){
                    // +=类名，表示在原有类的基础上，再添加一个类，表示标签有2个类
                    $('box').className += ' none';
                    isShow = false;
                }else{
                    $('box').className = 'box';
                    isShow = true;
                }
            }
        }
    </script>
</body>
</html>
```
